;Description
;-----------------------------------------------------------------------------
;Forces a refresh of the inventory menu.
;
;Silently adds, equips, unequips and removes an empty item to the Player's
;torch slot, and makes sure an already equipped torch stays equipped.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
ref CurrItem
;-----------------------------------------------------------------------------



;-----------------------------------------------------------------------------
begin Function { }
;-----------------------------------------------------------------------------

	Let CurrItem := Player.GetEquippedObject 14
	
	Player.AddItemNS DLMisc 1
	Player.EquipItemSilent DLMisc
	Player.UnEquipItemSilent DLMisc
	Player.RemoveItemNS DLMisc 1

	Player.EquipItem2NS CurrItem

	Return
;-----------------------------------------------------------------------------
end ;Function { }
