scn aaZefirosSetResists

;Main script
;-----------------------------------------------------------------------------
;aaZefirosManager
;-----------------------------------------------------------------------------

;Used scripts
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Related scripts
;-----------------------------------------------------------------------------
;aaZefirosDefaultINI
;-----------------------------------------------------------------------------

;Used globals
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Related globals
;-----------------------------------------------------------------------------
;aaZefirosRXPFire
;aaZefirosRXPFrost
;aaZefirosRXPShock
;aaZefirosRXPMagic
;aaZefirosRXPNormal
;aaZefirosRXPParalysis
;aaZefirosRXPDisease
;-----------------------------------------------------------------------------

;Description
;-----------------------------------------------------------------------------
;For enchantment in Enchs, adds Resist effects of the passed magnitudes.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
short Fire
short Frst
short Shck
short Magc
short Nrml
short Para
short Dise

ref Ench

array_var Item
array_var Enchs
;-----------------------------------------------------------------------------



;-----------------------------------------------------------------------------
begin Function { Fire, Frst, Shck, Magc, Nrml, Para, Dise, Enchs }
;-----------------------------------------------------------------------------

	ForEach Item <- Enchs

		Let Ench := *Item

		if ( Fire > 0 )
			
			AddFullEffectItem RSFI Fire 0 0 0 Ench
			
		endif
		
		if ( Frst > 0 )

			AddFullEffectItem RSFR Frst 0 0 0 Ench

		endif
		
		if ( Shck > 0 )

			AddFullEffectItem RSSH Shck 0 0 0 Ench

		endif
		
		if ( Magc > 0 )

			AddFullEffectItem RSMA Magc 0 0 0 Ench

		endif
		
		if ( Nrml > 0 )

			AddFullEffectItem RSNW Nrml 0 0 0 Ench

		endif
		
		if ( Para > 0 )

			AddFullEffectItem RSPA Para 0 0 0 Ench

		endif
		
		if ( Dise > 0 )

			AddFullEffectItem RSDI Dise 0 0 0 Ench

		endif

	loop

	
	Return
;-----------------------------------------------------------------------------
end ;Function { Fire, Frst, Shck, Magc, Nrml, Para, Dise, Enchs }
;-----------------------------------------------------------------------------
