;Related globals
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Description
;-----------------------------------------------------------------------------
;Given growth values for a MagicEffectItem and a level, calculate the current
;effect's stats and return them in a Map.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
short MagB
float MagM
short AreB
float AreM
short DurB
float DurM
short Lvl

long Magnitude
long Area
long Duration

array_var Stats ;Return value
;-----------------------------------------------------------------------------



;-----------------------------------------------------------------------------
begin Function { MagB, MagM, AreB, AreM, DurB, DurM, Lvl }
;-----------------------------------------------------------------------------

	Let Magnitude := MagB + ( MagM * Lvl )
	Let Area := AreB + ( AreM * Lvl )
	Let Duration := DurB + ( DurM * Lvl )

	Let Stats := ar_Map 0::Magnitude 1::Area 2::Duration


	SetFunctionValue Stats
	Return
;-----------------------------------------------------------------------------
end ;Function { MagB, MagM, AreB, AreM, DurB, DurM, Lvl }
;-----------------------------------------------------------------------------
