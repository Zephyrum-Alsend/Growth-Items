;Description

;Fires when a Player casted spell hits a target.
;Checks if the spell is a disease, then awards XP.
;
;Diseases are spells cast by the victim on themselves.  Even though diseases
;are acquired by getting hit by infected.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
ref MagicItem
ref Caster
;-----------------------------------------------------------------------------

;-----------------------------------------------------------------------------
begin Function { MagicItem, Caster }
;-----------------------------------------------------------------------------

	if ( 1 == GetSpellType MagicItem )

		DebugPrint "%n contracted %n." Caster MagicItem
		Let aaZefirosRXPDisease += aaZefirosInit.fXPDisease

	endif

end ;Function { MagicItem, Caster }
