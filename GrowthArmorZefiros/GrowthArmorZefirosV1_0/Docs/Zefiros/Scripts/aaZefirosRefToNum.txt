;Description
;-----------------------------------------------------------------------------
;Castes a Ref to a Long.
;Not supported directly by OBSE, but indirectly by converting Ref to 
;String_Var, then String_Var to Long.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
ref R
string_var Str
long L
;-----------------------------------------------------------------------------



;-----------------------------------------------------------------------------
begin Function { R }
;-----------------------------------------------------------------------------

	Let Str := sv_Construct "%i" R
	Let L := #Str

	
	SetFunctionValue L
	sv_Destruct Str
	Return
;-----------------------------------------------------------------------------
end ;Function { R }
;-----------------------------------------------------------------------------
