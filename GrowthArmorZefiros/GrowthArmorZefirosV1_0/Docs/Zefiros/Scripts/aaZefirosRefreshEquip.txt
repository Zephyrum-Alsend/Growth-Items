;Description
;-----------------------------------------------------------------------------
;Given an array of Items and an Actor, equip them on the Actor.
;Will fail if called on the same frame as UnEquip due to an engine bug.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
ref Actor
ref Item

array_var Items
array_var Iter
;-----------------------------------------------------------------------------



;-----------------------------------------------------------------------------
begin Function { Actor, Items }
;-----------------------------------------------------------------------------

	ForEach Iter <- Items

		Let Item := *Iter
		
		DebugPrint "Equipping %n on %n." Item Actor
		Actor.EquipItem2NS Item
			
	loop


	Return
;-----------------------------------------------------------------------------
end ;Function { Actor, Items }
;-----------------------------------------------------------------------------
