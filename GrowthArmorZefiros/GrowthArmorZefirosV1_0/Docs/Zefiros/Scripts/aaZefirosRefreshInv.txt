;Description
;-----------------------------------------------------------------------------
;Forces a refresh of the inventory menu.
;
;Silently adds, equips, unequips and removes an empty item to the Player's
;torch slot, and makes sure an already equipped torch stays equipped.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
ref Torch
;-----------------------------------------------------------------------------



;-----------------------------------------------------------------------------
begin Function { }
;-----------------------------------------------------------------------------

	Let Torch := Player.GetEquippedObject 14
	
	Player.AddItemNS DLRefr 1
	Player.EquipItemSilent DLRefr
	Player.UnEquipItemSilent DLRefr
	Player.RemoveItemNS DLRefr 1

	Player.EquipItem2NS Torch

	Return
;-----------------------------------------------------------------------------
end ;Function { }
