;-----------------------------------------------------------------------------

;Used globals
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Related globals
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Description
;-----------------------------------------------------------------------------
;Given an array of Items and an Actor, if the Actor has the item equipped, 
;re-equip.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
ref Actor
ref Item

array_var Items
array_var Iter
;-----------------------------------------------------------------------------



;-----------------------------------------------------------------------------
begin Function { Actor, Items }
;-----------------------------------------------------------------------------

	ForEach Iter <- Items

		Let Item := Iter["value"]
		
		if ( Actor.GetEquipped Item )

			DebugPrint "Re-equipping %n on %n." Item Actor
			Player.UnequipItemSilent Item
			Player.EquipItem2NS Item

		endif

	loop

	
	Return
;-----------------------------------------------------------------------------
end ;Function { Actor, Items }
;-----------------------------------------------------------------------------
