scn aaZefirosSetEquipHandlers

;Main script
;-----------------------------------------------------------------------------
;aaZefirosManager
;-----------------------------------------------------------------------------

;Used scripts
;-----------------------------------------------------------------------------
;aaZefirosOnUnEquip
;-----------------------------------------------------------------------------

;Related scripts
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Used globals
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Related globals
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Description
;-----------------------------------------------------------------------------
;Given an Actor and an array of Items, create OnUnEquip handlers for them.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
ref Target
array_var Items

ref Item
array_var Value
;-----------------------------------------------------------------------------

;-----------------------------------------------------------------------------
begin Function { Target, Items }
;-----------------------------------------------------------------------------

	ForEach Value <- Items

		Let Item := *Value

		;OnActorEquip fires twice for enchanted items, see Object% for OnEquip block
		SetEventHandler "OnActorUnequip" aaZefirosOnUnEquip first::Target second::Item

	loop

	
	Return
;-----------------------------------------------------------------------------
end ;Function { Target, Items }
;-----------------------------------------------------------------------------
