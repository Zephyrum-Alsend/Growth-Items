;Description
;-----------------------------------------------------------------------------
;Depending on Flag, appends the passed PreFix and/or PostFix to Name and 
;returns the resulting string.
;1 = Omit prefix
;2 = Omit postfix
;3 = Omit both
;else omit neither
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
string_var PreFix
string_var PostFix
short Flag
string_var Name ;Return value
;-----------------------------------------------------------------------------

;-----------------------------------------------------------------------------
begin Function { PreFix, PostFix, Flag, Name }
;-----------------------------------------------------------------------------

	;Omit PreFix
	if ( Flag == 1 )

		Let Name := Name + " " + PostFix

	;Omit PostFix
	elseif ( Flag == 2 )

		Let Name := PreFix + " " + Name

	;Omit neither
	elseif ( Flag != 3 )

		Let Name := Prefix + " " + Name + " " + PostFix

	endif
	

	SetFunctionValue Name
	Return
;-----------------------------------------------------------------------------
end ;Function { PreFix, PostFix, Flag, Name }
;-----------------------------------------------------------------------------
