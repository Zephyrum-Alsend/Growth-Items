;Description
;-----------------------------------------------------------------------------
;Menu for weight change.
;
;Called when Menu[0] = 8 || 9.
;
;Menu:
;0::MenuIndex
;1::Button
;2::Page
;3::SubMenu
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
string_var Name
array_var Menu ;Return value
;-----------------------------------------------------------------------------

;-----------------------------------------------------------------------------
begin Function { Name, Menu }
;-----------------------------------------------------------------------------

	if eval( Menu[1] == -2 )

		Let Name += " Weight:"
		MessageBox $Name "[Cancel]" "[<Back]" "Light" "Heavy"
		Let Menu[1] := -1
		
	elseif eval( Menu[1] < 0 )

		Let Menu[1] := GetButtonPressed

	elseif eval( Menu[1] == 1 )

		;Back
		DebugPrint "Returning to FormMenu."
		Let Menu[0] := 1
		Let Menu[1] := -2

	elseif eval( Menu[1] == 2 )

		;Light
		Let Menu[1] := 0
		Let Menu[3] := 1

	elseif eval( Menu[1] == 3 )

		;Heavy
		Let Menu[1] := 1
		Let Menu[3] := 1

	else

		;Cancel
		DebugPrint "Cancelling WeightMenu."
		Let Menu[0] := 0

	endif

	
	SetFunctionValue Menu
	Return
;-----------------------------------------------------------------------------
end ;Function { Name, Menu }
;-----------------------------------------------------------------------------
