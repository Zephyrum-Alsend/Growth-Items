scn aaZefirosSetEquipHandlers

;Main script

;aaZefirosManager
;-----------------------------------------------------------------------------

;Used scripts
;-----------------------------------------------------------------------------
;aaZefirosOnEquip
;aaZefirosOnUnEquip
;-----------------------------------------------------------------------------

;Related scripts
;-----------------------------------------------------------------------------

;-----------------------------------------------------------------------------

;Used globals
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Related globals
;-----------------------------------------------------------------------------
;-----------------------------------------------------------------------------

;Description
;-----------------------------------------------------------------------------
;Given an Actor and an array of Items, create OnEquip and OnUnEquip handlers 
;for them.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
ref Target
ref Item
array_var Items
array_var Value
;-----------------------------------------------------------------------------

;-----------------------------------------------------------------------------
begin Function { Target, Items }
;-----------------------------------------------------------------------------

	ForEach Value <- Items

		Let Item := Value["value"]

		;OnActorEquip fires twice for enchanted items, see Object for OnEquip block
		;SetEventHandler "OnActorEquip" aaZefirosOnEquip first::Target second::Item
		SetEventHandler "OnActorUnequip" aaZefirosOnUnEquip first::Target second::Item

	loop

	
	Return
;-----------------------------------------------------------------------------
end ;Function { Target, Items }
;-----------------------------------------------------------------------------
