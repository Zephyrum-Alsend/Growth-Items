;Description
;-----------------------------------------------------------------------------
;GetPCBipedIconPath
;Given a biped item (i.e. armor) returns a string containing the icon path of
;the same sex as the PC.
;-----------------------------------------------------------------------------

;Parameters
;-----------------------------------------------------------------------------
string_var Icon ;Return value
ref Biped
;-----------------------------------------------------------------------------

;-----------------------------------------------------------------------------
begin Function { Biped }
;-----------------------------------------------------------------------------

	if ( GetPCIsSex Male )

		Let Icon := GetBipedIconPath 0 Biped
		DebugPrint "Male%B BipedIconPath: %b%z" Icon
		
	else

		Let Icon := GetBipedIconPath 1 Biped
		DebugPrint "Female%B BipedIconPath: %b%z" Icon

	endif


	SetFunctionValue Icon
	sv_Destruct Icon
	Return
;-----------------------------------------------------------------------------
end ;Function { Biped }
